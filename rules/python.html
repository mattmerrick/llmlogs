1. Type Hints and Protocol Classes

Implement comprehensive type hints using Python's typing module and Protocol classes for better code clarity and maintainability. This modern approach to Python typing ensures better IDE support, easier refactoring, and fewer runtime errors.

python-type-hints.mdc 
from typing import Protocol, TypeVar, List, Optional
from dataclasses import dataclass
from datetime import datetime

# Define a Protocol for database operations
class DatabaseProtocol(Protocol):
    def save(self, data: dict) -> bool: ...
    def find_by_id(self, id: str) -> Optional[dict]: ...
    def update(self, id: str, data: dict) -> bool: ...

# Define generic type for models
T = TypeVar('T')

@dataclass
class BaseModel:
    id: str
    created_at: datetime
    updated_at: datetime

@dataclass
class User(BaseModel):
    email: str
    username: str
    is_active: bool = True

class Repository(Generic[T]):
    def __init__(self, db: DatabaseProtocol, model_class: Type[T]):
        self.db = db
        self.model_class = model_class
    
    def find_by_id(self, id: str) -> Optional[T]:
        data = self.db.find_by_id(id)
        return self.model_class(**data) if data else None
    
    def save(self, model: T) -> bool:
        return self.db.save(asdict(model))

# Example usage with type checking
class MongoDBAdapter:
    def save(self, data: dict) -> bool:
        # Implementation for MongoDB
        return True
    
    def find_by_id(self, id: str) -> Optional[dict]:
        # Implementation for MongoDB
        return {"id": id, "email": "user@example.com"}
    
    def update(self, id: str, data: dict) -> bool:
        # Implementation for MongoDB
        return True

# The type checker will ensure this is valid
db: DatabaseProtocol = MongoDBAdapter()
user_repo = Repository[User](db, User)

# Type-safe operations
user = user_repo.find_by_id("123")
if user:
    print(f"Found user: {user.email}")  # IDE autocomplete works here

# Error caught by type checker
user.invalid_field  # Type error caught by IDE/mypy

Why This Rule Matters

  • Enables better IDE support and code completion
  • Catches type-related errors before runtime
  • Makes code more maintainable and self-documenting
  • Facilitates easier refactoring
  • Improves collaboration in large codebases
