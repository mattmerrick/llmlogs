About These Rules

These TypeScript coding rules are specifically formatted for the Cursor Editor as .mdc files, focusing on TypeScript 5.0+ features and advanced type patterns. Each rule is designed to help you write more type-safe code while leveraging Cursor's AI capabilities for enhanced development.

1. Advanced Type Utilities and Inference

Implement advanced type utilities and leverage TypeScript's powerful type inference system for better code maintainability and type safety. This approach demonstrates modern TypeScript patterns including conditional types, mapped types, and template literal types.

typescript-advanced-types.mdc 
// Type-safe API response handler with discriminated unions
type ApiResponse = 
  | { status: 'success'; data: T; }
  | { status: 'error'; error: { code: number; message: string; }; }
  | { status: 'loading'; }

// Template literal type for valid API endpoints
type ApiEndpoint = `/api/v1/${string}`

// Utility type for making all properties optional at any depth
type DeepPartial = {
    [P in keyof T]?: T[P] extends object ? DeepPartial : T[P];
};

// Type-safe record validator using mapped types
type Validator = {
    [P in keyof T]: (value: T[P]) => boolean;
};

// Example usage in a type-safe API client
interface User {
    id: string;
    email: string;
    preferences: {
        theme: 'light' | 'dark';
        notifications: boolean;
    };
}

class ApiClient {
    async fetchData(endpoint: ApiEndpoint): Promise> {
        try {
            const response = await fetch(endpoint);
            const data = await response.json();
            return { status: 'success', data };
        } catch (error) {
            return {
                status: 'error',
                error: { code: 500, message: error.message }
            };
        }
    }

    // Type-safe partial updates
    async updateUser(
        userId: string,
        updates: DeepPartial
    ): Promise> {
        return this.fetchData(`/api/v1/users/${userId}`);
    }
}

// Type-safe form validation
const userValidator: Validator = {
    id: (value) => typeof value === 'string' && value.length > 0,
    email: (value) => /^[^@]+@[^@]+\.[^@]+$/.test(value),
    preferences: (value) => {
        return (
            typeof value.theme === 'string' &&
            ['light', 'dark'].includes(value.theme) &&
            typeof value.notifications === 'boolean'
        );
    }
};

// Example usage with type inference
const api = new ApiClient();

async function handleUserUpdate(userId: string) {
    // TypeScript infers all possible response types
    const response = await api.updateUser(userId, {
        preferences: { theme: 'dark' }
    });

    switch (response.status) {
        case 'success':
            // TypeScript knows response.data is User
            console.log(response.data.preferences.theme);
            break;
        case 'error':
            // TypeScript knows error structure
            console.error(response.error.message);
            break;
        case 'loading':
            // Handle loading state
            break;
    }
}
