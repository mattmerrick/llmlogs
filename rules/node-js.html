1. Async Programming Patterns

Implement modern async programming patterns for better performance and maintainability. This approach ensures better handling of asynchronous operations and takes advantage of Node.js's event-driven architecture.

nodejs-async-patterns.mdc 
// Async utility functions
const sleep = (ms) => new Promise(resolve => setTimeout(resolve, ms));

// Async error handling
class AsyncError extends Error {
    constructor(message, code, details = {}) {
        super(message);
        this.name = this.constructor.name;
        this.code = code;
        this.details = details;
        Error.captureStackTrace(this, this.constructor);
    }
}

// Async retry mechanism
async function withRetry(fn, options = {}) {
    const {
        maxAttempts = 3,
        delay = 1000,
        backoff = 2,
        onRetry = () => {}
    } = options;

    let lastError;
    for (let attempt = 1; attempt <= maxAttempts; attempt++) {
        try {
            return await fn();
        } catch (error) {
            lastError = error;
            if (attempt === maxAttempts) break;
            
            await sleep(delay * Math.pow(backoff, attempt - 1));
            onRetry(error, attempt);
        }
    }
    throw lastError;
}

// Example usage
class UserService {
    constructor(db) {
        this.db = db;
    }

    async getUser(id) {
        return withRetry(
            async () => {
                const user = await this.db.users.findById(id);
                if (!user) {
                    throw new AsyncError('User not found', 'NOT_FOUND', { id });
                }
                return user;
            },
            {
                maxAttempts: 3,
                delay: 1000,
                onRetry: (error, attempt) => {
                    console.log(`Retry attempt ${attempt} for user ${id}`);
                }
            }
        );
    }

    async updateUser(id, data) {
        return withRetry(
            async () => {
                const user = await this.db.users.update(id, data);
                if (!user) {
                    throw new AsyncError('Failed to update user', 'UPDATE_FAILED', { id, data });
                }
                return user;
            },
            {
                maxAttempts: 3,
                delay: 1000,
                onRetry: (error, attempt) => {
                    console.log(`Retry attempt ${attempt} for updating user ${id}`);
                }
            }
        );
    }
}

// Async middleware pattern
function asyncHandler(fn) {
    return (req, res, next) => {
        Promise.resolve(fn(req, res, next)).catch(next);
    };
}

// Example Express route with async handler
app.get('/users/:id', asyncHandler(async (req, res) => {
    const userService = new UserService(db);
    const user = await userService.getUser(req.params.id);
    res.json(user);
}));

  • Facilitates better performance
  • Reduces callback hell
  • Enables better debugging

2. Error Handling and Logging

Implement robust error handling and logging patterns for better debugging and maintenance. This approach helps Cursor's AI better understand error contexts and provide more accurate debugging assistance.

nodejs-error-handling.mdc 
// Custom error types
class AppError extends Error {
    constructor(message, code, details = {}) {
        super(message);
        this.name = this.constructor.name;
        this.code = code;
        this.details = details;
        Error.captureStackTrace(this, this.constructor);
    }
}

class ValidationError extends AppError {
    constructor(message, details) {
        super(message, 'VALIDATION_ERROR', details);
    }
}

class DatabaseError extends AppError {
    constructor(message, details) {
        super(message, 'DATABASE_ERROR', details);
    }
}

// Logger utility
const winston = require('winston');

const logger = winston.createLogger({
    level: process.env.LOG_LEVEL || 'info',
    format: winston.format.combine(
        winston.format.timestamp(),
        winston.format.json()
    ),
    transports: [
        new winston.transports.File({ filename: 'error.log', level: 'error' }),
        new winston.transports.File({ filename: 'combined.log' })
    ]
});

if (process.env.NODE_ENV !== 'production') {
    logger.add(new winston.transports.Console({
        format: winston.format.simple()
    }));
}

// Error handling middleware
function errorHandler(err, req, res, next) {
    logger.error('Error occurred:', {
        error: err.message,
        code: err.code,
        stack: err.stack,
        details: err.details
    });

    if (err instanceof AppError) {
        return res.status(getStatusCode(err.code)).json({
            error: err.message,
            code: err.code,
            details: err.details
        });
    }

    // Handle unexpected errors
    res.status(500).json({
        error: 'Internal server error',
        code: 'INTERNAL_ERROR'
    });
}

function getStatusCode(code) {
    const statusMap = {
        'VALIDATION_ERROR': 400,
        'AUTHENTICATION_ERROR': 401,
        'AUTHORIZATION_ERROR': 403,
        'NOT_FOUND': 404,
        'DATABASE_ERROR': 500
    };
    return statusMap[code] || 500;
}

// Example usage
async function validateUser(userData) {
    try {
        if (!userData.email) {
            throw new ValidationError('Email is required', {
                field: 'email',
                value: userData.email
            });
        }

        if (!userData.email.includes('@')) {
            throw new ValidationError('Invalid email format', {
                field: 'email',
                value: userData.email
            });
        }

        return true;
    } catch (error) {
        logger.error('User validation failed', {
            error: error.message,
            code: error.code,
            details: error.details
        });
        throw error;
    }
}

// Express route with error handling
app.post('/users', async (req, res, next) => {
    try {
        await validateUser(req.body);
        const user = await userService.createUser(req.body);
        res.status(201).json(user);
    } catch (error) {
        next(error);
    }
});

3. Module Organization

Implement proper module organization patterns for better code maintainability and performance. This approach helps Cursor's AI better understand code structure and provide more accurate suggestions.

nodejs-module-organization.mdc 
// config/index.js
const config = {
    port: process.env.PORT || 3000,
    database: {
        url: process.env.DATABASE_URL,
        options: {
            useNewUrlParser: true,
            useUnifiedTopology: true
        }
    },
    logging: {
        level: process.env.LOG_LEVEL || 'info',
        format: 'json'
    }
};

module.exports = config;

// utils/validation.js
const { ValidationError } = require('../errors');

function validateEmail(email) {
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    return emailRegex.test(email);
}

function validateUser(userData) {
    if (!userData.email) {
        throw new ValidationError('Email is required', {
            field: 'email',
            value: userData.email
        });
    }

    if (!validateEmail(userData.email)) {
        throw new ValidationError('Invalid email format', {
            field: 'email',
            value: userData.email
        });
    }

    return true;
}

module.exports = {
    validateEmail,
    validateUser
};

// services/userService.js
const { DatabaseError } = require('../errors');
const { validateUser } = require('../utils/validation');
const logger = require('../utils/logger');

class UserService {
    constructor(db) {
        this.db = db;
    }

    async createUser(userData) {
        try {
            validateUser(userData);
            const user = await this.db.users.create(userData);
            logger.info('User created successfully', { userId: user.id });
            return user;
        } catch (error) {
            logger.error('Failed to create user', {
                error: error.message,
                code: error.code,
                details: error.details
            });
            throw error;
        }
    }

    async getUser(id) {
        try {
            const user = await this.db.users.findById(id);
            if (!user) {
                throw new DatabaseError('User not found', { id });
            }
            return user;
        } catch (error) {
            logger.error('Failed to get user', {
                error: error.message,
                code: error.code,
                details: error.details
            });
            throw error;
        }
    }
}

module.exports = UserService;
